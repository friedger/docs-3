## Address balances

Retrieves a paginated list of address balances

### Query Parameters

<Property name={"offset"} type={"Offset"} required={false} deprecated={false}>

<span>Minimum: `0`</span>

</Property>

<Property name={"limit"} type={"Limit"} required={false} deprecated={false}>

<span>Minimum: `1`</span>

<span>Maximum: `60`</span>

</Property>

export type Address = string;

 * Rune balance
 */
export type Balance = string;

export interface PaginatedBalancesResponse {
  limit: number;
  offset: number;
  total: number;
  results: ({
    rune: RuneDetail;

    balance: Balance;
  })[];
}
/**
