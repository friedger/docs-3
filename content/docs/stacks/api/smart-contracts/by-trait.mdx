Retrieves a list of contracts based on the following traits listed in JSON format -  functions, variables, maps, fungible tokens and non-fungible tokens.

### Query parameters

<Property required={true} deprecated={false} name={"trait_abi"} type={"string"}>

JSON abi of the trait.

</Property>

<Property required={false} deprecated={false} name={"limit"} type={"integer"}>

Max number of contracts fetch

</Property>

<Property required={false} deprecated={false} name={"offset"} type={"integer"}>

Index of first contract event to fetch

</Property>

| Status code | Description |
| :----------- | :----------- |
| `200` | List of contracts implement given trait |

</div>

```ts
/**
 * GET list of contracts
 */
export interface ContractListResponse {
  /**
   * The number of contracts to return
   */
  limit: number;
  /**
   * The number to contracts to skip (starting at `0`)
   */
  offset: number;
  results: SmartContract[];
}
/**
 * A Smart Contract Detail
 */
export interface SmartContract {
  tx_id: string;
  canonical: boolean;
  contract_id: string;
  block_height: number;
  source_code: string;
  abi: string;
}

```
