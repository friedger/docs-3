```clarity
(define-read-only (get-tenure-height (block uint))
  (ok
    (at-block
      (unwrap!
        ;; !hover get-stacks-block-info
        (get-stacks-block-info? id-header-hash block)
        ;; !hover error
        (err u404)
      )
      ;; !hover tenure-height
      tenure-height
    )
  )
)
```
